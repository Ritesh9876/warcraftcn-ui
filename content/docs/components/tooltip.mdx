---
title: Tooltip
description: A popup that displays Warcraft-styled information when the element receives keyboard focus or the mouse hovers over it.
---

import { Button } from "@/components/ui/warcraftcn/button";
import {
  Tooltip,
  TooltipBody,
  TooltipContent,
  TooltipTitle,
  TooltipTrigger,
} from "@/components/ui/warcraftcn/tooltip";

<Tooltip>
  <TooltipTrigger asChild>
    <Button className="w-64 md:w-96">Hover</Button>
  </TooltipTrigger>
  <TooltipContent>
    <TooltipTitle>Hearthstone</TooltipTitle>
    <TooltipBody>Binds when picked up</TooltipBody>
  </TooltipContent>
</Tooltip>

### Side

Use the `side` prop to change the position of the tooltip.

<div className="flex flex-wrap gap-2">
  <Tooltip>
    <TooltipTrigger asChild>
      <Button>Left</Button>
    </TooltipTrigger>
    <TooltipContent side="left">
      <p>Left side</p>
    </TooltipContent>
  </Tooltip>

<Tooltip>
  <TooltipTrigger asChild>
    <Button>Top</Button>
  </TooltipTrigger>
  <TooltipContent side="top">
    <p>Top side</p>
  </TooltipContent>
</Tooltip>

<Tooltip>
  <TooltipTrigger asChild>

## Installation

```bash
pnpm dlx shadcn@latest add https://warcraftcn.com/r/tooltip.json
```

## Usage

```tsx
import {
  Tooltip,
  TooltipBody,
  TooltipContent,
  TooltipTitle,
  TooltipTrigger,
} from "@/components/ui/warcraftcn/tooltip";
```

```tsx
<Tooltip>
  <TooltipTrigger>Hover</TooltipTrigger>
  <TooltipContent>
    <TooltipTitle>Item Name</TooltipTitle>
    <TooltipBody>Item description</TooltipBody>
  </TooltipContent>
</Tooltip>
```
